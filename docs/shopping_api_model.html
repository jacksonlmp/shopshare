BROADCAST item.checked Todos veem o item riscado em tempo real
"event": "item.checked", "payload": { "item_id": "item-uuid", "is_checked": true, "checked_by": { "display_name": "Maria" }, "checked_at": "2024-01-15T11:30:00Z" }
